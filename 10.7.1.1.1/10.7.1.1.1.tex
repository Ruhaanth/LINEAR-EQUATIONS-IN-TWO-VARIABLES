\begin{enumerate}
\item Find the distance between the following pairs of points.\\
 i)(2, 3), (4, 1) \\
\solution:\\
Given,
\begin{align}
\vec{A}&=\myvec{2\\3}\\
\vec{B}&=\myvec{4\\1}\\
\vec{AB}&=\myvec{-2\\2}\\
|AB|&=\sqrt{(-2)^2+(2)^2}\\
|AB|&=\sqrt{4+4}\\
|AB|&=\sqrt{8}\\
|AB|&=2\sqrt{2}\\
\end{align}
\end{enumerate}
